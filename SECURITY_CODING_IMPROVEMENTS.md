### 1. SECURITY.PY - SECRET_KEY –∏ PASSWORD VALIDATION

**–¢–µ–∫—É—â–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ:** ‚ùå SECRET_KEY –∑–∞—Ö–∞—Ä–¥–∫–æ–∂–µ–Ω, –ø–∞—Ä–æ–ª–∏ —Å–ª–∞–±—ã–µ

**–§–∞–π–ª:** `backend/utils/security.py`

**–î–µ–π—Å—Ç–≤–∏–µ - –ü–æ–ª–Ω–æ—Å—Ç—å—é –ø–µ—Ä–µ–ø–∏—Å–∞—Ç—å:**

```python
import os
import logging
from passlib.context import CryptContext
from jose import jwt, JWTError
from datetime import datetime, timedelta
from pydantic import validator

logger = logging.getLogger("slideconfirm")

# ‚úÖ –ù–û–í–û–ï: –ü–æ–ª—É—á–∞—Ç—å SECRET_KEY –∏–∑ .env
SECRET_KEY = os.getenv("SECRET_KEY")
if not SECRET_KEY:
    raise ValueError("‚ùå FATAL: SECRET_KEY –Ω–µ —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω –≤ .env!")

ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = int(os.getenv("ACCESS_TOKEN_EXPIRE", 30))
REFRESH_TOKEN_EXPIRE_DAYS = int(os.getenv("REFRESH_TOKEN_EXPIRE", 7))

# ‚úÖ –ù–û–í–û–ï: bcrypt —Å 12+ rounds –≤–º–µ—Å—Ç–æ 10
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto", bcrypt__rounds=12)

def hash_password(password: str) -> str:
    """–•–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ –ø–∞—Ä–æ–ª—è —Å bcrypt"""
    return pwd_context.hash(password)

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–∞—Ä–æ–ª—è"""
    return pwd_context.verify(plain_password, hashed_password)

# ‚úÖ –ù–û–í–û–ï: Token blacklist –¥–ª—è logout
TOKEN_BLACKLIST = set()  # –í –ø—Ä–æ–¥–∞–∫—à–µ–Ω–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å Redis

def add_to_blacklist(token: str):
    """–î–æ–±–∞–≤–∏—Ç—å —Ç–æ–∫–µ–Ω –≤ —á–µ—Ä–Ω—ã–π —Å–ø–∏—Å–æ–∫"""
    TOKEN_BLACKLIST.add(token)
    logger.warning(f"Token added to blacklist")

def is_token_blacklisted(token: str) -> bool:
    """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å, –≤ —á–µ—Ä–Ω–æ–º –ª–∏ —Å–ø–∏—Å–∫–µ —Ç–æ–∫–µ–Ω"""
    return token in TOKEN_BLACKLIST

def create_access_token(data: dict, expires_delta: timedelta = None):
    """–°–æ–∑–¥–∞—Ç—å access token (30 –º–∏–Ω—É—Ç)"""
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES))
    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

def create_refresh_token(data: dict):
    """–°–æ–∑–¥–∞—Ç—å refresh token (7 –¥–Ω–µ–π)"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=REFRESH_TOKEN_EXPIRE_DAYS)
    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

def decode_access_token(token: str):
    """–î–µ–∫–æ–¥–∏—Ä–æ–≤–∞—Ç—å —Ç–æ–∫–µ–Ω"""
    try:
        # ‚úÖ –ù–û–í–û–ï: –ü—Ä–æ–≤–µ—Ä–∫–∞ blacklist
        if is_token_blacklisted(token):
            logger.warning(f"‚ùå Blacklisted token used")
            return None
        
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        return payload
    except JWTError as e:
        logger.warning(f"‚ùå JWT Error: {str(e)}")
        return None

def validate_password_strength(password: str) -> bool:
    """‚úÖ –ù–û–í–û–ï: –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Å—Ç–æ–π–∫–æ—Å—Ç—å –ø–∞—Ä–æ–ª—è"""
    if len(password) < 12:
        raise ValueError("–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å –º–∏–Ω–∏–º—É–º 12 —Å–∏–º–≤–æ–ª–æ–≤")
    
    if not any(c.isupper() for c in password):
        raise ValueError("–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ó–ê–ì–õ–ê–í–ù–´–ï –±—É–∫–≤—ã (A-Z)")
    
    if not any(c.islower() for c in password):
        raise ValueError("–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Å—Ç—Ä–æ—á–Ω—ã–µ –±—É–∫–≤—ã (a-z)")
    
    if not any(c.isdigit() for c in password):
        raise ValueError("–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ü–∏—Ñ—Ä—ã (0-9)")
    
    if not any(c in "!@#$%^&*()_+-=[]{}|;:,.<>?" for c in password):
        raise ValueError("–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Å–ø–µ—Ü—Å–∏–º–≤–æ–ª—ã (!@#$%^&*)")
    
    return True
```

### 4. AUTH.PY - Rate Limiting –∏ –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ

**–¢–µ–∫—É—â–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ:** ‚ùå –ù–µ—Ç rate limiting, –Ω–µ—Ç –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è

**–§–∞–π–ª:** `backend/auth.py`

**–î–µ–π—Å—Ç–≤–∏–µ - –ü–æ–ª–Ω–æ—Å—Ç—å—é –ø–µ—Ä–µ–ø–∏—Å–∞—Ç—å:**

```python

from fastapi import APIRouter, Depends, HTTPException, status, Request
from sqlalchemy.orm import Session
from .database import get_db
from .models import User
from .schemas import UserCreate, UserLogin, TokenResponse, RefreshTokenRequest
from .utils.security import (
    hash_password, 
    verify_password, 
    create_access_token,
    create_refresh_token,
    decode_access_token,
    validate_password_strength,
    add_to_blacklist
)
from slowapi import Limiter
from slowapi.util import get_remote_address

router = APIRouter(prefix="/auth", tags=["auth"])
limiter = Limiter(key_func=get_remote_address)
logger = logging.getLogger("slideconfirm")

# ‚úÖ –ù–û–í–û–ï: –¢—Ä–µ–∫–∏–Ω–≥ –Ω–µ—É–¥–∞—á–Ω—ã—Ö –ø–æ–ø—ã—Ç–æ–∫ –ª–æ–≥–∏–Ω–∞
failed_login_attempts = {}  # {email: [(timestamp, ip), ...]}

@router.post("/register", response_model=UserResponse)
@limiter.limit("5/hour")  # ‚úÖ –ú–∞–∫—Å–∏–º—É–º 5 —Ä–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏–π –≤ —á–∞—Å
def register(request: Request, user: UserCreate, db: Session = Depends(get_db)):
    """–†–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    existing = db.query(User).filter(User.email == user.email).first()
    if existing:
        logger.warning(f"‚ùå Registration failed: Email already registered - {user.email} from {request.client.host}")
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST, 
            detail="Email already registered"
        )
    
    try:
        # ‚úÖ –ù–û–í–û–ï: –í–∞–ª–∏–¥–∞—Ü–∏—è –ø–∞—Ä–æ–ª—è (—É–∂–µ –≤ Pydantic, –Ω–æ –ø—Ä–æ–≤–µ—Ä—è–µ–º –µ—â–µ —Ä–∞–∑)
        validate_password_strength(user.password)
    except ValueError as e:
        logger.warning(f"‚ùå Registration failed: Weak password - {user.email}")
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(e))
    
    # –•–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ –ø–∞—Ä–æ–ª—è
    hashed_password = hash_password(user.password)
    role = getattr(user, 'role', 'user') or 'user'
    
    # –°–æ–∑–¥–∞–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    db_user = User(
        first_name=user.first_name,
        last_name=user.last_name,
        email=user.email,
        password_hash=hashed_password,
        role=role
    )
    
    try:
        db.add(db_user)
        db.commit()
        db.refresh(db_user)
        logger.info(f"‚úÖ Registration successful: {user.email}")
        return db_user
    except Exception as e:
        db.rollback()
        logger.error(f"‚ùå Registration error: {str(e)}")
        raise HTTPException(status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, detail="Registration failed")

@router.post("/login", response_model=TokenResponse)
@limiter.limit("5/15 minutes")  # ‚úÖ 5 –ø–æ–ø—ã—Ç–æ–∫ –∑–∞ 15 –º–∏–Ω—É—Ç
def login(request: Request, user: UserLogin, db: Session = Depends(get_db)):
    """–õ–æ–≥–∏–Ω –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
    
    # –ü–æ–∏—Å–∫ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    db_user = db.query(User).filter(User.email == user.email).first()
    if not db_user:
        logger.warning(f"‚ùå LOGIN FAILED: User not found - {user.email} from {request.client.host}")
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED, 
            detail="Invalid credentials"
        )
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–∞—Ä–æ–ª—è
    if not verify_password(user.password, db_user.password_hash):
        logger.warning(f"‚ùå LOGIN FAILED: Invalid password - {user.email} from {request.client.host}")
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED, 
            detail="Invalid credentials"
        )
    
    # ‚úÖ –ù–û–í–û–ï: –°–æ–∑–¥–∞–Ω–∏–µ –æ–±–æ–∏—Ö —Ç–æ–∫–µ–Ω–æ–≤
    access_token = create_access_token(data={"sub": str(db_user.id), "role": db_user.role})
    refresh_token = create_refresh_token(data={"sub": str(db_user.id)})
    
    logger.info(f"‚úÖ LOGIN SUCCESS: {user.email} from {request.client.host}")
    
    return TokenResponse(
        access_token=access_token,
        refresh_token=refresh_token,
        role=db_user.role
    )

# ‚úÖ –ù–û–í–û–ï: Endpoint –¥–ª—è refresh token
@router.post("/refresh", response_model=TokenResponse)

def refresh_access_token(request: Request, req: RefreshTokenRequest, db: Session = Depends(get_db)):
```

```javascript

  failedQueue.forEach(prom => {
    if (error) {
      prom.reject(error);
    } else {
      prom.resolve(token);
    }
  });
  
  isRefreshing = false;
  failedQueue = [];

  (response) => response,
  async (error) => {
    const originalRequest = error.config;

          failedQueue.push({ resolve, reject });
        }).then(token => {
          originalRequest.headers.Authorization = `Bearer ${token}`;
          return apiClient(originalRequest);
        });

      originalRequest._retry = true;
      isRefreshing = true;

        sessionStorage.setItem('token', access_token);
        
        apiClient.defaults.headers.common.Authorization = `Bearer ${access_token}`;
        originalRequest.headers.Authorization = `Bearer ${access_token}`;

        return apiClient(originalRequest);
      } catch (err) {
        processQueue(err, null);
        sessionStorage.clear();
        window.location.href = '/login';
        return Promise.reject(err);
      }
    }
    
    return Promise.reject(error);
  }
```
